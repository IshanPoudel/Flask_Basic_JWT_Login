46def calculate_mse_for_timeframe(model, features, target, data, start_date, end_date):
47 subset_data = data[(data.index >= start_date) & (data.index <= end_date)]
48 predictions = model.predict(subset_data[features])
49 mse = mean_squared_error(subset_data[target], predictions)
50 return mse, subset_data.index, subset_data[target], predictions
